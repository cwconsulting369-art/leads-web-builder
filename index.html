function render() {

}

function toggle(id) {
  const l = LEADS.find(x=>x.id===id); if (!l||l._gen) return;
  if (selected.has(id)) selected.delete(id); else selected.add(id);
  render();
}

function toggleAll() {
  const e = visible.filter(l=>!l._gen);

  render();
}

async function genSelectedDirect(force=false) {
  const toGen = LEADS.filter(l=>selected.has(l.id)); if (!toGen.length) return;
  document.getElementById('gen-btn').disabled=true;
  document.getElementById('gen-force').disabled=true;
  document.getElementById('gen-force').textContent='⏳';
  for (const lead of toGen) {
    if (!force) {
      const cached = memCache.find(m=>m.maps===lead.maps||m.name?.toLowerCase()===lead.name?.toLowerCase());
      if (cached?.html) {
        lead._cached=true; lead._done=true; lead._html=cached.html;
        selected.delete(lead.id); render();
        genHTML=cached.html; curName=lead.name; showPreview(cached.html);
        setStatus(`✦ ${lead.name} aus Cache`,`p`); continue;
      }
    }
    lead._gen=true; selected.delete(lead.id); render();
    setStatus(`KI generiert: ${lead.name}...`,`a`);
    const ctx = buildMemCtx(lead.cat);
    const html = await genLeadKI(lead, ctx);
    lead._gen=false;
    if (html) {
      lead._done=true; lead._html=html; genHTML=html; curName=lead.name;
      showPreview(html);
      memCache.push({name:lead.name,maps:lead.maps,cat:lead.cat,html,notes:`KI · ${lead.reviews} Reviews ${lead.rating}★`});
      updateLearnStats(); setStatus(`✓ ${lead.name} (KI)`,`g`);
    } else setStatus(`⚠ Fehler bei ${lead.name}`);
    render();
  }
  document.getElementById('gen-force').textContent='⚡ KI';
  document.getElementById('gen-force').disabled=selected.size===0;
  document.getElementById('gen-btn').disabled=selected.size===0;
}

function buildMemCtx(cat) {
  const r = memCache.filter(m=>m.cat===cat).slice(-3);
  return r.length ? `\n\nLERN-KONTEXT:\n${r.map(m=>`- ${m.name}: ${m.notes||'generiert'}`).join('\n')}\nBaue auf diesen Erfahrungen auf.` : '';
}

async function genLeadKI(lead, ctx='') {
  showLoading();
  const p = `Du bist ein Top-Webdesigner.\n\nName: ${lead.name}\nBranche: ${lead.cat}\nAdresse: ${lead.address}, ${lead.plz} ${lead.city}\nRating: ${lead.rating}⭐ (${lead.reviews} Reviews)\nMaps: ${lead.maps}${ctx}\n\nNutze web_search für Reviews und Infos.\n\nVollständige One-Page-Website als HTML: Hero → Über uns → Leistungen → Testimonials → Kontakt.\nNur HTML, beginne mit .`;
  try {
    // Use proxy endpoint (CORS fix)
    const endpoint = '/api/generate';
    let res = await fetch(endpoint, {
      method:'POST', headers:{'Content-Type':'application/json'},
      body: JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:8000,tools:[{type:'web_search_20250305',name:'web_search'}],messages:[{role:'user',content:p}]})
    });
    let data = await res.json();
    if (data.stop_reason==='tool_use') {
      const tr = data.content.filter(b=>b.type==='tool_use').map(b=>({type:'tool_result',tool_use_id:b.id,content:'done'}));
      const r2 = await fetch(endpoint, {
        method:'POST', headers:{'Content-Type':'application/json'},
        body: JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:8000,tools:[{type:'web_search_20250305',name:'web_search'}],messages:[{role:'user',content:p},{role:'assistant',content:data.content},{role:'user',content:tr}]})
      });
      data = await r2.json();
    }
    if (data.error) { showError('API: '+data.error.message); return null; }
    let html = data.content?.filter(b=>b.type==='text').map(b=>b.text).join('')||'';
    html = html.replace(/^```html\s*/i,'').replace(/\s*```\s*$/i,'').trim();
    return html && html.includes(' `
      

        
${escH(m.name)}

        
${m.cat||'friseur'}${escH(m.notes||'')}

      

    `).join('');
  }
  updateLearnStats();
  setStatus(`${memCache.length} Memory-Einträge`, memCache.length?'g':'');
}
